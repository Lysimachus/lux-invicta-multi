antitolerant_ruler = {
	name = CB_NAME_TOLERANT_RULER
	war_name = WAR_NAME_TOLERANT_RULER
	sprite = 11
	truce_days = 3650
	
	is_permanent = yes
	can_ask_to_join_war = no

	allowed_to_target_tributaries = no
	allowed_to_target_suzerains = no

	can_use = {
		ROOT = {
			trait = sectarian_heavy #Sectarian rulers won't tolerate tolerant ones.
			custom_tooltip = {
				text = NOT_BUSY
				hidden_tooltip = {
					is_inaccessible_trigger = no
				}
			}
		}
		FROM = {
			trait = tolerant_ruler
			NOT = { religion = catholic } #the Pope decides if the realm is worthy of excommunication
			religion_group = ROOT
			NOT = { 
				any_current_enemy = {
					NOT = {
						religion_group = ROOT
					}
				}
			}
		}
	}

	can_use_gui = {
		ROOT = {
			piety = 100
		}
	}


	is_valid = {
		FROM = {
			trait = tolerant_ruler
			religion_group = ROOT
		}
	}

	on_success = {
		ROOT = {
			if = {
				limit = {
					FROM = {
						primary_title = {
							tier = BARON
						}
					}
				}
				ROOT = {
					piety = 10
					prestige = 50
				}
			}
			if = {
				limit = {
					FROM = {
						primary_title = {
							tier = COUNT
						}
					}
				}
				ROOT = {
					piety = 20
					prestige = 100
				}
			}
			if = {
				limit = {
					FROM = {
						primary_title = {
							tier = DUKE
						}
					}
				}
				ROOT = {
					piety = 30
					prestige = 150
				}
			}
			if = {
				limit = {
					FROM = {
						primary_title = {
							tier = KING
						}
					}
				}
				ROOT = {
					piety = 50
					prestige = 200
				}
			}
			if = {
				limit = {
					FROM = {
						primary_title = {
							tier = EMPEROR
						}
					}
				}
				ROOT = {
					piety = 70
					prestige = 400
				}
			}
		}

		FROM = {
			remove_trait = tolerant_ruler
			any_demesne_title = { add_law = tolerance_1 }
			prestige = -200
			abdicate = yes
		}
	}

	on_fail = {
		ROOT = {
			piety = -150
			prestige = -100
			religion_authority = {
				modifier = doubted_faith
				years = 100
			}
		}
	}

	on_reverse_demand = {
		ROOT = {
			piety = -100
			prestige = -200
			transfer_scaled_wealth = {
				to = FROM
				value = 4.0
			}
		}
		FROM = {
			prestige = 200
		}
	}


	attacker_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	defender_ai_defeat_worth = {
		factor = 100
	}
	ai_will_do = { 
		factor = 1
		
		modifier = {
			factor = 2.5
			ROOT = { controls_religion = yes } #Caliphate and Fylkirate strike back
			}
		modifier = {
			factor = 0.25
			ROOT = { trait = cynical }
			}
		modifier = {
			factor = 1.5
			ROOT = { religion_group = christian }
			}
		modifier = {
			factor = 1.25
			ROOT = { religion_group = muslim }
			}
		modifier = {
			factor = 0.15
			ROOT = { religion_group = indian_group }
			}
		modifier = {
			factor = 0.25
			ROOT = { religion_group = zoroastrian_group }
			}
		modifier = {
			factor = 0.5
			ROOT = { religion_group = graecoroman_group }
			}
		modifier = {
			factor = 0.5
			ROOT = { is_reformed_religion = no }
			}
		}
}

lift_excommunication = {
	name = CB_NAME_LIFT_EXCOMMUNICATION
	war_name = WAR_NAME_LIFT_EXCOMMUNICATION
	sprite = 11
	truce_days = 3650
	is_permanent = yes
	is_holy_war = yes
	can_ask_to_join_war = yes
	attacker_can_call_allies = no
	defender_can_call_allies = yes
	
	can_use = {
		ROOT = {
			religion = FROM
			trait = excommunicated
			rightful_religious_head_scope = { ROOT = { excommunicated_for = PREV } }
			religion = catholic
			rightful_religious_head = FROM
			custom_tooltip = {
				text = NOT_BUSY
				hidden_tooltip = {
					is_inaccessible_trigger = no
				}
			}
		}
		FROM = { controls_religion = yes }
	}
	
	is_valid = {
		ROOT = {
			religion = FROM
			trait = excommunicated
			rightful_religious_head_scope = { ROOT = { excommunicated_for = PREV } }
			religion = catholic
			rightful_religious_head = FROM
		}
		FROM = { controls_religion = yes }
	}
	
	on_add = { ROOT = { character_event = { id = relf.15 } }
		}
	on_success = {
		any_attacker = {
			limit = { character = ROOT }
			participation_scaled_piety = -500
			participation_scaled_prestige = -250
		}
		any_attacker = {
			limit = {
				NOT = {
					character = ROOT
				}
				religion = ROOT
			}
			hidden_tooltip = { 
				participation_scaled_piety = -500
				participation_scaled_prestige = -250
			}
		}
		ROOT = {
			religion_authority = {
				modifier = forced_lifting
				years = 50
			}
			rightful_religious_head_scope = {
				opinion = {
					who = ROOT
					modifier = opinion_forced_lifting
				}
			}
			character_event = { id = relf.14 } #the Pope receives the ultimatum
		}
		
		FROM = { 
			piety = -500
			prestige = -250
			
		}
	}

	on_fail = {
		ROOT = { 
			piety = -250
			prestige = -125
		}
		FROM = { 
			piety = 100
			prestige = 50
		}
	}

	on_reverse_demand = {
		ROOT = {
			transfer_scaled_wealth = {
				to = FROM
				value = 4.0
			}
			piety = -500
			prestige = -250
		}
		
		FROM = { 
			piety = 500
			prestige = 250
		}
	}

	attacker_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	defender_ai_defeat_worth = {
		factor = 100
	}
}
